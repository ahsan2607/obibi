\section{Pendahuluan \& Gambaran Umum}

% --- SLIDE: LATAR BELAKANG ---
\begin{frame}{Latar Belakang}
    \begin{itemize}
        \item Banyak pasien kronis, lansia, dan penyandang disabilitas kesulitan mengikuti jadwal pengobatan.
        \item Keterbatasan fisik, kognitif, dan akses teknologi menjadi kendala utama.
        \item Dibutuhkan solusi digital yang \textbf{inklusif, mudah digunakan, dan cerdas}.
    \end{itemize}
    \vspace{0.5cm}
    \begin{block}{Solusi: Aplikasi Obibi}
        Asisten kesehatan berbasis \textbf{Agentic AI} dan \textbf{GraphRAG} untuk:
        \begin{itemize}
            \item Pengingat obat otomatis dan personal.
            \item Analisis interaksi antar obat secara mandiri.
            \item Chatbot AI dengan dukungan perintah suara.
        \end{itemize}
    \end{block}
\end{frame}

% --- SLIDE: TUJUAN PRODUK ---
\begin{frame}{Tujuan \& Manfaat Produk}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Tujuan Utama:}
        \begin{itemize}
            \item Meningkatkan kepatuhan pasien terhadap pengobatan.
            \item Menyediakan pengingat yang terpersonalisasi dan inklusif.
            \item Memberikan akses informasi kesehatan yang aman dan akurat.
        \end{itemize}
        
        \column{0.5\textwidth}
        \textbf{Manfaat:}
        \begin{itemize}
            \item \textbf{Pasien}: Pengingat otomatis, asisten virtual 24/7.
            \item \textbf{Tenaga Medis}: Membantu memantau kepatuhan pasien.
            \item \textbf{Ekosistem Kesehatan}: Model sistem kesehatan inklusif berbasis AI.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: PROFIL PENGGUNA ---
\begin{frame}{Profil \& Kelas Pengguna}
    \begin{itemize}
        \item \textbf{Pasien (Utama)}:
        \begin{itemize}
            \item Lansia dan penyandang disabilitas.
            \item Pasien kronis dengan pengobatan rutin.
            \item Kemampuan teknis rendah hingga menengah.
        \end{itemize}
        \item \textbf{Medical Admin / Apoteker (Sekunder)}:
        \begin{itemize}

% --- SLIDE: USE CASE SCENARIO #1 ---
\begin{frame}{Use Case: Registrasi Pengguna Baru}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Calon pengguna membuka halaman register.
            \item Mengisi data diri (Nama, Email, Kata Sandi).
            \item Menekan tombol "Register".
            \item Sistem memvalidasi data dan menyimpan ke database.
            \item Pengguna diarahkan ke halaman login.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item Email sudah terdaftar: Sistem menampilkan peringatan.
            \item Koneksi internet terputus: Sistem menampilkan pesan error koneksi.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Akun pengguna berhasil terdaftar di database.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: USE CASE SCENARIO #2 ---
\begin{frame}{Use Case: Login Pengguna}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Pengguna membuka halaman login.
            \item Memasukkan Email dan Kata Sandi yang valid.
            \item Menekan tombol "Login".
            \item Sistem memvalidasi kredensial pengguna.
            \item Pengguna diarahkan ke dashboard utama.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item Kredensial tidak valid: Sistem menampilkan pesan kesalahan login.
            \item Koneksi terputus: Tampil pesan kesalahan jaringan.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Sesi pengguna aktif dibuat dan masuk ke dashboard.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: USE CASE SCENARIO #3 ---
\begin{frame}{Use Case: Kelola Data Obat}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Pasien membuka menu Kelola Obat.
            \item Klik Tambah Obat.
            \item Isi form (Nama, Deskripsi, Dosis, Stok, Bentuk).
            \item Klik Simpan Obat.
            \item Sistem validasi data dan simpan ke database.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item Input kosong / tidak valid: sistem menampilkan pesan kesalahan.
            \item Hapus obat: Pasien mengonfirmasi dialog hapus, sistem menghapus obat beserta jadwal terkait secara cascade.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Data obat terupdate di database dan muncul di daftar.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: USE CASE SCENARIO #4 ---
\begin{frame}{Use Case: Input Jadwal Minum Obat}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Pasien buka halaman Kelola Obat.
            \item Klik Tambah Jadwal.
            \item Isi form (Tipe, Waktu, Tanggal).
            \item Klik Simpan Jadwal.
            \item Sistem validasi dan simpan ke database.
            \item \textit{Cronjob} reminder aktif.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item Validasi gagal: tampilkan pesan error.
            \item Database error: notifikasi gagal simpan.
            \item Cronjob gagal: log error, retry otomatis.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Jadwal tersimpan, cronjob aktif.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: USE CASE SCENARIO #5 ---
\begin{frame}{Use Case: Cek Interaksi Antar Obat}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Pasien membuka menu interaksi.
            \item Pasien mengecek interaksi antar obat.
            \item Sistem AI mengecek data jurnal klinis (GraphRAG).
            \item Sistem mencari konteks dan membuat kesimpulan.
            \item Pasien menerima hasil analisis \& rekomendasi.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item Jika obat tidak ditemukan, sistem menyarankan konsultasi dokter.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Pasien melihat kesimpulan interaksi obat.
        \end{itemize}
    \end{columns}
\end{frame}

% --- SLIDE: USE CASE SCENARIO #6 ---
\begin{frame}{Use Case: Chat dengan Asisten AI (Obibi)}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \textbf{Skenario Normal:}
        \begin{enumerate}
            \item Pasien membuka halaman Chat.
            \item Mengetik pertanyaan medis / merekam perintah suara.
            \item Menekan kirim.
            \item AI Engine memproses pertanyaan dengan GraphRAG.
            \item Sistem menampilkan respon asisten virtual kesehatan.
        \end{enumerate}
        
        \column{0.5\textwidth}
        \textbf{Alternatif:}
        \begin{itemize}
            \item AI Engine lambat / timeout: Menampilkan pesan fallback untuk mencoba lagi.
            \item Suara tidak jelas (STT): Meminta pengguna mengulangi perintah suara.
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Post-Condition:}
        \begin{itemize}
            \item Riwayat percakapan pasien tersimpan.
        \end{itemize}
    \end{columns}
\end{frame}

        \item Sistem dibangun dengan arsitektur yang modular dan skalabel, siap untuk dikembangkan lebih lanjut.
